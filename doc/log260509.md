```python
import sys
import csv
import time
from datetime import datetime

import serial
import serial.tools.list_ports

from PySide6.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, QHBoxLayout,
    QPushButton, QComboBox, QLabel, QSlider, QFileDialog
)
from PySide6.QtCore import QTimer, Qt

import pyqtgraph as pg


class ESP32ControlUI(QWidget):
    def __init__(self):
        super().__init__()

        self.setWindowTitle("ESP32 Sensor Control UI")
        self.resize(900, 600)

        self.ser = None
        self.rows = []

        self.time_data = []
        self.temp_data = []

        self.build_ui()

        self.timer = QTimer()
        self.timer.timeout.connect(self.read_serial_data)
        self.timer.start(20)

    def build_ui(self):
        main_layout = QVBoxLayout()

        # ---------- Serial control ----------
        serial_layout = QHBoxLayout()

        self.port_combo = QComboBox()
        self.refresh_ports()

        self.refresh_button = QPushButton("重新整理 Port")
        self.refresh_button.clicked.connect(self.refresh_ports)

        self.connect_button = QPushButton("連線")
        self.connect_button.clicked.connect(self.toggle_connection)

        serial_layout.addWidget(QLabel("Serial Port:"))
        serial_layout.addWidget(self.port_combo)
        serial_layout.addWidget(self.refresh_button)
        serial_layout.addWidget(self.connect_button)

        main_layout.addLayout(serial_layout)

        # ---------- Command buttons ----------
        command_layout = QHBoxLayout()

        self.start_button = QPushButton("START")
        self.start_button.clicked.connect(lambda: self.send_command("START"))

        self.stop_button = QPushButton("STOP")
        self.stop_button.clicked.connect(lambda: self.send_command("STOP"))

        self.export_button = QPushButton("匯出 CSV")
        self.export_button.clicked.connect(self.export_csv)

        command_layout.addWidget(self.start_button)
        command_layout.addWidget(self.stop_button)
        command_layout.addWidget(self.export_button)

        main_layout.addLayout(command_layout)

        # ---------- PWM slider ----------
        pwm_layout = QHBoxLayout()

        self.pwm_label = QLabel("PWM: 0")

        self.pwm_slider = QSlider(Qt.Horizontal)
        self.pwm_slider.setMinimum(0)
        self.pwm_slider.setMaximum(255)
        self.pwm_slider.setValue(0)
        self.pwm_slider.valueChanged.connect(self.update_pwm)

        pwm_layout.addWidget(self.pwm_label)
        pwm_layout.addWidget(self.pwm_slider)

        main_layout.addLayout(pwm_layout)

        # ---------- Latest data ----------
        self.data_label = QLabel("尚未收到資料")
        main_layout.addWidget(self.data_label)

        # ---------- Plot ----------
        self.plot_widget = pg.PlotWidget()
        self.plot_widget.setTitle("Temperature")
        self.plot_widget.setLabel("left", "Temp", units="C")
        self.plot_widget.setLabel("bottom", "Time", units="s")

        self.temp_curve = self.plot_widget.plot(self.time_data, self.temp_data)

        main_layout.addWidget(self.plot_widget)

        self.setLayout(main_layout)

    def refresh_ports(self):
        self.port_combo.clear()

        ports = serial.tools.list_ports.comports()

        for port in ports:
            self.port_combo.addItem(port.device)

    def toggle_connection(self):
        if self.ser is None:
            port = self.port_combo.currentText()

            if not port:
                self.data_label.setText("找不到 Serial Port")
                return

            try:
                self.ser = serial.Serial(port, 115200, timeout=0.01)
                self.connect_button.setText("斷線")
                self.data_label.setText(f"已連線到 {port}")
            except Exception as e:
                self.data_label.setText(f"連線失敗: {e}")

        else:
            self.ser.close()
            self.ser = None
            self.connect_button.setText("連線")
            self.data_label.setText("已斷線")

    def send_command(self, command):
        if self.ser is not None and self.ser.is_open:
            self.ser.write((command + "\n").encode("utf-8"))
        else:
            self.data_label.setText("尚未連線，無法送出指令")

    def update_pwm(self, value):
        self.pwm_label.setText(f"PWM: {value}")
        self.send_command(f"SET_PWM {value}")

    def read_serial_data(self):
        if self.ser is None:
            return

        try:
            while self.ser.in_waiting:
                line = self.ser.readline().decode("utf-8", errors="ignore").strip()

                if not line:
                    continue

                self.handle_line(line)

        except Exception as e:
            self.data_label.setText(f"讀取錯誤: {e}")

    def handle_line(self, line):
        # ESP32 傳來：
        # DATA,12345,26.4,0.01,-0.02,9.81

        parts = line.split(",")

        if parts[0] == "DATA" and len(parts) == 6:
            esp_time_ms = int(parts[1])
            temp = float(parts[2])
            ax = float(parts[3])
            ay = float(parts[4])
            az = float(parts[5])

            pc_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S.%f")[:-3]

            row = {
                "pc_time": pc_time,
                "esp_time_ms": esp_time_ms,
                "temp": temp,
                "ax": ax,
                "ay": ay,
                "az": az,
            }

            self.rows.append(row)

            self.data_label.setText(
                f"Temp: {temp:.2f} °C | "
                f"ax: {ax:.3f}, ay: {ay:.3f}, az: {az:.3f}"
            )

            t = esp_time_ms / 1000.0
            self.time_data.append(t)
            self.temp_data.append(temp)

            # 只顯示最近 500 筆，避免圖越跑越慢
            self.time_data = self.time_data[-500:]
            self.temp_data = self.temp_data[-500:]

            self.temp_curve.setData(self.time_data, self.temp_data)

        elif parts[0] == "ACK":
            self.data_label.setText(f"ESP32 回覆: {line}")

        elif parts[0] == "ERR":
            self.data_label.setText(f"ESP32 錯誤: {line}")

        else:
            self.data_label.setText(f"未知資料: {line}")

    def export_csv(self):
        if not self.rows:
            self.data_label.setText("沒有資料可以匯出")
            return

        filename, _ = QFileDialog.getSaveFileName(
            self,
            "儲存 CSV",
            "esp32_log.csv",
            "CSV Files (*.csv)"
        )

        if not filename:
            return

        with open(filename, "w", newline="", encoding="utf-8-sig") as f:
            writer = csv.DictWriter(
                f,
                fieldnames=["pc_time", "esp_time_ms", "temp", "ax", "ay", "az"]
            )

            writer.writeheader()
            writer.writerows(self.rows)

        self.data_label.setText(f"已匯出: {filename}")


if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = ESP32ControlUI()
    window.show()
    sys.exit(app.exec())
```

## 韌體
燒入esp32

```C++
const int LED_PIN = 2;

bool running = false;
int pwmValue = 0;

unsigned long lastSendTime = 0;
const unsigned long sendInterval = 50; // 50 ms = 20 Hz

void setup() {
  Serial.begin(115200);

  pinMode(LED_PIN, OUTPUT);

  // 如果你有感測器，例如 MPU6050 / BME280 / GPS
  // 在這裡初始化
}

void loop() {
  readCommandFromPC();

  if (running) {
    sendSensorData();
  }
}

void readCommandFromPC() {
  if (Serial.available()) {
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();

    if (cmd == "START") {
      running = true;
      Serial.println("ACK,START");
    }
    else if (cmd == "STOP") {
      running = false;
      Serial.println("ACK,STOP");
    }
    else if (cmd.startsWith("SET_PWM")) {
      int spaceIndex = cmd.indexOf(' ');
      if (spaceIndex > 0) {
        pwmValue = cmd.substring(spaceIndex + 1).toInt();
        pwmValue = constrain(pwmValue, 0, 255);

        // 這裡可以換成控制馬達、伺服、LED、繼電器等
        analogWrite(LED_PIN, pwmValue);

        Serial.print("ACK,SET_PWM,");
        Serial.println(pwmValue);
      }
    }
    else {
      Serial.print("ERR,UNKNOWN_CMD,");
      Serial.println(cmd);
    }
  }
}

void sendSensorData() {
  unsigned long now = millis();

  if (now - lastSendTime >= sendInterval) {
    lastSendTime = now;

    // 這裡先用假資料，你之後換成實際感測器資料
    float temp = 26.4;
    float ax = 0.01;
    float ay = -0.02;
    float az = 9.81;

    Serial.print("DATA,");
    Serial.print(now);
    Serial.print(",");
    Serial.print(temp, 3);
    Serial.print(",");
    Serial.print(ax, 3);
    Serial.print(",");
    Serial.print(ay, 3);
    Serial.print(",");
    Serial.println(az, 3);
  }
}
```
